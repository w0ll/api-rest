CREATE TABLE "Cliente" (
	"ID"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"Nome"	VARCHAR(50),
	"DataNascimento"	DATE,
	"Documento"	VARCHAR(14)
);

CREATE TABLE "ItensVendidos" (
	"ID"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"Produto"	INTEGER,
	"Venda"	INTEGER,
	"Quantidade"	DECIMAL(10 , 2)
);

CREATE TABLE "Produto" (
	"ID"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"Nome"	VARCHAR(50),
	"Valor"	DECIMAL(10 , 2)
);

CREATE TABLE "Venda" (
	"ID"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"DataVenda"	DATE,
	"Cliente"	INTEGER
);
